```shell
sudo perl -pi -e 's/\x80\x79\x05\x00\x0F\x94\xC2/\xC6\x41\x05\x01\xB2\x00\x90/' /Applications/Sublime\ Text.app/Contents/MacOS/sublime_text

sudo codesign --sign - --force --deep /Applications/Sublime\ Text.app
```
